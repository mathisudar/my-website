JavaScript if

Display "Good day!" if the hour is less than 18:00:

Good Evening!

Example Code


                    
                <script>

                if (new Date().getHours() < 18) { 
                    document.getElementById("demo9a").innerHTML="Good day!" ; 
                }
                
                </script>

JavaScript if .. else

A time-based greeting:

Example Code


                            
                        <script>
        
                        const hour = new Date().getHours();
                        let greeting;
                        
                        if (hour < 18) { 
                            greeting="Good day" ; 
                        } 
                        else { 
                            greeting="Good evening" ; 
                        }
                            
                        document.getElementById("demo9b").innerHTML=greeting;
                        
                        </script>

JavaScript if .. else

A time-based greeting:

Example Code


                                    
                <script>

                const time = new Date().getHours();
                let greeting9c;
                
                if (time < 10) { 
                    greeting9c="Good morning" ; 
                } 

                else if (time < 20) { 
                    greeting9c="Good day" ; 
                } 

                else {
                    greeting9c="Good evening" ; 
                } 
                
                document.getElementById("demo9c").innerHTML=greeting9c;
                
                </script>

JavaScript Control Flow

The switch Statement

Example Code


                                        
                    <script>
    
                    let day;
                    let date = new Date().getDay();
                    
                    switch (date) {
                    
                    case 0:
                    day = "Sunday";
                    break;
                    
                    case 1:
                    day = "Monday";
                    break;
                    
                    case 2:
                    day = "Tuesday";
                    break;
                    
                    case 3:
                    day = "Wednesday";
                    break;
                    
                    case 4:
                    day = "Thursday";
                    break;
                    
                    case 5:
                    day = "Friday";
                    break;
                    
                    case 6:
                    day = "Saturday";
                    
                    }
                    
                    document.getElementById("demo9d").innerHTML = "Today is " + day;

                    </script>
